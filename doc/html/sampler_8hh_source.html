218  template <class Space>
219  double sampler<Space>::GetESS(void) const
220  {
221  long double sum = 0;
222  long double sumsq = 0;
223 
224  for(int i = 0; i < N; i++)
225  sum += expl(pParticles[i].GetLogWeight());
226 
227  for(int i = 0; i < N; i++)
228  sumsq += expl(2.0*(pParticles[i].GetLogWeight()));
229 
230  return expl(-log(sumsq) + 2*log(sum));
231  }

263  {
264  long double rValue = 0;
265  long double wSum = 0;
266  for(int i =0; i < N; i++)

268  rValue += expl(pParticles[i].GetLogWeight()) * pIntegrand(pParticles[i].GetValue(), pAuxiliary);

270  }
271 
272  rValue /= wSum;

323  template <class Space>
324  double sampler<Space>::IterateEss(void)
325  {
326  //Initially, the current particle set should be appended to the historical process.
327  if(htHistoryMode != SMC_HISTORY_NONE)
328  History.Push(N, pParticles, nAccepted, historyflags(nResampled));
329 
330  nAccepted = 0;
331 
332  //Move the particle set.
333  MoveParticles();
334 
335  //Normalise the weights to sensible values....
336  double dMaxWeight = -std::numeric_limits<double>::infinity();
337  for(int i = 0; i < N; i++)
338  dMaxWeight = max(dMaxWeight, pParticles[i].GetLogWeight());
339  for(int i = 0; i < N; i++)
340  pParticles[i].SetLogWeight(pParticles[i].GetLogWeight() - (dMaxWeight));
341 
342 
343  //Check if the ESS is below some reasonable threshold and resample if necessary.
344  //A mechanism for setting this threshold is required.
345  double ESS = GetESS();
346  if(ESS < dResampleThreshold) {
347  nResampled = 1;
348  Resample(rtResampleMode);
349  }
350  else
351  nResampled = 0;
352  //A possible MCMC step should be included here.
353  for(int i = 0; i < N; i++) {
354  if(Moves.DoMCMC(T+1,pParticles[i], pRng))
355  nAccepted++;
356  }
357  // Increment the evolution time.
358  T++;
359 
360  return ESS;
361  }

379  template <class Space>
380  void sampler<Space>::Resample(ResampleType lMode)
381  {
382  //Resampling is done in place.
383  double dWeightSum = 0;
384  unsigned uMultinomialCount;
385 
386  //First obtain a count of the number of children each particle has.
387  switch(lMode) {
388  case SMC_RESAMPLE_MULTINOMIAL:
389  //Sample from a suitable multinomial vector
390  for(int i = 0; i < N; ++i)
391  dRSWeights[i] = pParticles[i].GetWeight();
392  pRng->Multinomial(N,N,dRSWeights,uRSCount);
393  break;
394 
395  case SMC_RESAMPLE_RESIDUAL:
396  //Sample from a suitable multinomial vector and add the integer replicate
397  //counts afterwards.
398  dWeightSum = 0;
399  for(int i = 0; i < N; ++i) {
400  dRSWeights[i] = pParticles[i].GetWeight();
401  dWeightSum += dRSWeights[i];
402  }
403 
404  uMultinomialCount = N;
405  for(int i = 0; i < N; ++i) {
406  dRSWeights[i] = N*dRSWeights[i] / dWeightSum;
407  uRSIndices[i] = unsigned(floor(dRSWeights[i])); //Reuse temporary storage.
408  dRSWeights[i] = (dRSWeights[i] - uRSIndices[i]);
409  uMultinomialCount -= uRSIndices[i];
410  }
411  pRng->Multinomial(uMultinomialCount,N,dRSWeights,uRSCount);
412  for(int i = 0; i < N; ++i)
413  uRSCount[i] += uRSIndices[i];
414  break;
415 
416 
417  case SMC_RESAMPLE_STRATIFIED:
418  default:
419  {
420  // Procedure for stratified sampling
421  dWeightSum = 0;
422  double dWeightCumulative = 0;
423  // Calculate the normalising constant of the weight vector
424  for(int i = 0; i < N; i++)
425  dWeightSum += exp(pParticles[i].GetLogWeight());
426  //Generate a random number between 0 and 1/N times the sum of the weights
427  double dRand = pRng->Uniform(0,1.0 / ((double)N));
428 
429  int j = 0, k = 0;
430  for(int i = 0; i < N; ++i)
431  uRSCount[i] = 0;
432 
433  dWeightCumulative = exp(pParticles[0].GetLogWeight()) / dWeightSum;
434  while(j < N) {
435  while((dWeightCumulative - dRand) > ((double)j)/((double)N) && j < N) {
436  uRSCount[k]++;
437  j++;
438  dRand = pRng->Uniform(0,1.0 / ((double)N));
439  }
440  k++;
441  dWeightCumulative += exp(pParticles[k].GetLogWeight()) / dWeightSum;
442  }
443  break;
444  }
445 
446  case SMC_RESAMPLE_SYSTEMATIC:
447  {
448  // Procedure for stratified sampling but with a common RV for each stratum
449  dWeightSum = 0;
450  double dWeightCumulative = 0;
451  // Calculate the normalising constant of the weight vector
452  for(int i = 0; i < N; i++)
453  dWeightSum += exp(pParticles[i].GetLogWeight());
454  //Generate a random number between 0 and 1/N times the sum of the weights
455  double dRand = pRng->Uniform(0,1.0 / ((double)N));
456 
457  int j = 0, k = 0;
458  for(int i = 0; i < N; ++i)
459  uRSCount[i] = 0;
460 
461  dWeightCumulative = exp(pParticles[0].GetLogWeight()) / dWeightSum;
462  while(j < N) {
463  while((dWeightCumulative - dRand) > ((double)j)/((double)N) && j < N) {
464  uRSCount[k]++;
465  j++;
466 
467  }
468  k++;
469  dWeightCumulative += exp(pParticles[k].GetLogWeight()) / dWeightSum;
470  }
471  break;
472 
473  }
474  }
475 
476  //Map count to indices to allow in-place resampling
477  for (unsigned int i=0, j=0; i<N; ++i) {
478  if (uRSCount[i]>0) {
479  uRSIndices[i] = i;
480  while (uRSCount[i]>1) {
481  while (uRSCount[j]>0) ++j; // find next free spot
482  uRSIndices[j++] = i; // assign index
483  --uRSCount[i]; // decrement number of remaining offsprings
484  }
485  }
486  }
487 
488  //Perform the replication of the chosen.
489  for(int i = 0; i < N ; ++i) {
490  if(uRSIndices[i] != i)
491  pParticles[i].SetValue(pParticles[uRSIndices[i]].GetValue());
492  pParticles[i].SetLogWeight(0);
493  }
494  }
