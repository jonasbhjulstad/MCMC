180  template <class Particle>
181  double historyelement<Particle>::GetESS(void) const
182  {
183  double sum = 0;
184  double sumsq = 0;
185 
186  for(int i = 0; i < number; i++) {
187  sum += exp(value[i].GetLogWeight());
188  sumsq += exp(2.0*value[i].GetLogWeight());
189  }
190  return (sum*sum)/sumsq;
191  }

198  long double historyelement<Particle>::Integrate(long lTime, double (*pIntegrand)(long,const Particle&,void*), void* pAuxiliary)
199  {
200  long double rValue = 0;
201  long double wSum = 0;
202 
203  for(int i =0; i < number; i++)
204  {
205  rValue += expl(value[i].GetLogWeight()) * (long double)pIntegrand(lTime, value[i], pAuxiliary);
206  wSum += expl(value[i].GetLogWeight());
207  }
208 
209 
210  rValue /= wSum;
211 
212  return rValue;
213  }
